114   private static Map<String, Object> expected() {

118   private static Map<String, Object> expected(byte[] dataBufferBytes, byte[] fileBytes) {
119     Map<String, Object> map = new LinkedHashMap<>();
120     map.put(FORM_FIELD_NAME, FORM_FIELD_VALUE);
121     map.put(FILE_PART_NAME, FILE_PART_CONTENT);
122     if (!isEmpty(dataBufferBytes)) {
123       map.put(DATA_BUFFER_PART_NAME, toString(dataBufferBytes));
124     }
125     if (!isEmpty(fileBytes)) {
126       map.put(REAL_FILES_PART_NAME, toString(fileBytes));
127     }
128     return map;
129   }

131   private static byte[] randomBytes(int len) {
132     Random random = new Random();
133     byte[] bytes = new byte[len];
134     random.nextBytes(bytes);
135     return bytes;
136   }
137 
138   private static String toString(byte[] bytes) {
139     return Base64.getEncoder().encodeToString(bytes);
140   }
141 
142   private Flux<DataBuffer> toDataBuffer(byte[] bytes, int bufferSize) {
143     return DataBufferUtils.read(
144         new ByteArrayResource(bytes),
145         new DefaultDataBufferFactory(),
146         bufferSize);
147   }

175   /**
176    * Delete tmp file.
177    */
178   @AfterEach
179   void deleteTmpFile() {
180     if (nonNull(tmpFile)) {
181       try {
182         Files.delete(tmpFile);
183         tmpFile = null;
184       } catch (IOException e) {
185         throw new IllegalStateException("Deleting tmp file failed.", e);
186       }
187     }
188   }
189 
190   private void writeToTmpFile(byte[] bytes) {
191     try {
192       tmpFile = Files.createTempFile("apiclient", "java");
193       Files.copy(new ByteArrayInputStream(bytes), tmpFile, StandardCopyOption.REPLACE_EXISTING);
194 
195     } catch (IOException e) {
196       throw new IllegalStateException("Creating tmp file failed.", e);
197     }
198   }

201    * Post multipart data map with web client.

203   @Test
204   void postMultipartDataMapWithWebClient() {
205     MultipartBodyBuilder builder = new MultipartBodyBuilder();
206     builder.part(FORM_FIELD_NAME, FORM_FIELD_VALUE, MediaType.TEXT_PLAIN);
207     builder.part(FILE_PART_NAME, new ClassPathResource(FILE_PART_RESOURCE));
208     StepVerifier
209         .create(webClient
210             .post()
211             .uri("/api/multipart/map")
212             .contentType(MediaType.MULTIPART_FORM_DATA)
213             .body(BodyInserters.fromMultipartData(builder.build()))
214             .retrieve()
215             .bodyToMono(new MapRef()))
216         .assertNext(response -> assertThat(response)
217             .containsExactlyInAnyOrderEntriesOf(expected()))
218         .expectNextCount(0)
219         .verifyComplete();
220   }

222   /**
223    * Post multipart data map.
224    */
225   @Test
226   void postMultipartDataMap() {
227     MultiValueMap<String, Part> partMap = new LinkedMultiValueMap<>();
228     partMap.add(
229         FORM_FIELD_NAME,
230         partBuilder.part(FORM_FIELD_NAME, FORM_FIELD_VALUE).contentType(MediaType.TEXT_PLAIN)
231             .build());
232     partMap.add(
233         FILE_PART_NAME,
234         partBuilder.part(FILE_PART_NAME, new ClassPathResource(FILE_PART_RESOURCE)).build());
235 
236     byte[] dataBuf = randomBytes(16 * 1024 + 8);
237     partMap.add(
238         DATA_BUFFER_PART_NAME,
239         partBuilder.part(DATA_BUFFER_PART_NAME, toDataBuffer(dataBuf, 256))
240             .contentType(MediaType.APPLICATION_OCTET_STREAM)
241             .build());
242 
243     byte[] fileBytes = randomBytes(3000);
244     writeToTmpFile(fileBytes);
245     partMap.add(
246         REAL_FILES_PART_NAME,
247         partBuilder.part(REAL_FILES_PART_NAME, tmpFile)
248             .contentType(MediaType.APPLICATION_OCTET_STREAM)
249             .build());
250 
251     StepVerifier.create(apiClient.postMultipartDataMap(partMap))
252         .assertNext(response -> assertThat(response)
253             .containsExactlyInAnyOrderEntriesOf(expected(dataBuf, fileBytes)))
254         .expectNextCount(0)
255         .verifyComplete();
256   }

409   @Test
410   void postFluxParts() {
411     List<Part> parts = new ArrayList<>();
412     parts.add(
413         partBuilder.part(FORM_FIELD_NAME, FORM_FIELD_VALUE).contentType(MediaType.TEXT_PLAIN)
414             .build());
415     parts.add(
416         partBuilder.part(FILE_PART_NAME, new ClassPathResource(FILE_PART_RESOURCE)).build());
417 
418     byte[] dataBuf = randomBytes(7 * 1024 + 7);
419     parts.add(
420         partBuilder.part(DATA_BUFFER_PART_NAME, toDataBuffer(dataBuf, 128))
421             .contentType(MediaType.APPLICATION_OCTET_STREAM)
422             .build());
423 
424     byte[] fileBytes = randomBytes(1234);
425     writeToTmpFile(fileBytes);
426     parts.add(
427         partBuilder.part(REAL_FILES_PART_NAME, tmpFile)
428             .contentType(MediaType.APPLICATION_OCTET_STREAM)
429             .build());
430 
431     StepVerifier.create(apiClient.postFluxParts(Flux.fromStream(parts.stream())))
432         .assertNext(response -> assertThat(response)
433             .containsExactlyInAnyOrderEntriesOf(expected(dataBuf, fileBytes)))
434         .expectNextCount(0)
435         .verifyComplete();
436
